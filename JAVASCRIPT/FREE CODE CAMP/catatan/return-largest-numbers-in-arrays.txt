function largestOfFour(arr) {

  let newArr = []
  for (let i = 0; i < arr.length; i++) {
    let max = -i
    for (let j = 0; j < arr[i].length; j++) {
      if (arr[i][j] > max) {
        max = arr[i][j]
      }
    }
    newArr.push(max)
  }
  console.log(newArr)
  return newArr;
}

largestOfFour([[13, 27, 18, 26], [4, 5, 1, 3], [32, 35, 37, 39], [1000, 1001, 857, 1]]);
